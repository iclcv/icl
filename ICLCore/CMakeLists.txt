file(GLOB HEADERS src/ICLCore/*.h)

  set_source_files_properties(src/ICLCore/OpenCV.cpp PROPERTIES COMPILE_FLAGS -Wno-narrowing)
endif()

  SOURCES ${SOURCES}
  HEADERS ${HEADERS}
