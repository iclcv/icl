210  inline std::vector<int> get_random_index_subset(int containerSize,
211  int subsetSize){
212  if(subsetSize > containerSize){
213  throw ICLException("get_random_index_subset: subsetsize must be <= containerSize");
214  }
215  std::vector<int> s(containerSize);
216  for(int i=0;i<containerSize;++i) s[i] = i;
217  std::random_shuffle(s.begin(), s.end());
218  return std::vector<int>(s.begin(), s.begin()+subsetSize);
219  }
