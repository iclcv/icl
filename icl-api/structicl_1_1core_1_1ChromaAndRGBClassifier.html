bool is_pixel_skin_colored(int r, int g, int b, ChromaClassifier c, int refcol[3], int threshold[3]){
return c(r,g,b)
&& abs(r-refcol[0])<threshold[0]
&& abs(g-refcol[1])<threshold[1]
&& abs(b-refcol[2])<threshold[2];
}
