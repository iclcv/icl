210  template<class T,unsigned int N> static inline T power(const T&x){
211  switch(N){
212  case 0: return 1;
213  case 1: return x;
214  case 2: return sqr(x);
215  case 3: return sqr(x)*x;
216  case 4: return sqr(sqr(x));
217  case 5: return sqr(sqr(x))*x;
218  default:
219  return ::pow(x,(int)N);
220  }
