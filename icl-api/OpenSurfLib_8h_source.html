126  class ICLCV_API ResponseLayer{
127  public:
128 
129  int width, height, step, filter;
130  float *responses;
131  unsigned char *laplacian;
132 
133  inline ResponseLayer(int width, int height, int step, int filter){
134  assert(width > 0 && height > 0);
135 
136  this->width = width;
137  this->height = height;
138  this->step = step;
139  this->filter = filter;
140 
141  responses = new float[width*height];
142  laplacian = new unsigned char[width*height];
143 
144  memset(responses,0,sizeof(float)*width*height);
145  memset(laplacian,0,sizeof(unsigned char)*width*height);
146  }
147 
148  inline ~ResponseLayer(){
149  if (responses) delete [] responses;
150  if (laplacian) delete [] laplacian;
151  }
152 
153  inline unsigned char getLaplacian(unsigned int row, unsigned int column)
154  {
155  return laplacian[row * width + column];
156  }
157 
158  inline unsigned char getLaplacian(unsigned int row, unsigned int column, ResponseLayer *src)
159  {
160  int scale = this->width / src->width;
161 
162 #ifdef RL_DEBUG
163  assert(src->getCoords(row, column) == this->getCoords(scale * row, scale * column));
164 #endif
165 
166  return laplacian[(scale * row) * width + (scale * column)];
167  }
168 
169  inline float getResponse(unsigned int row, unsigned int column)
170  {
171  return responses[row * width + column];
172  }
173 
174  inline float getResponse(unsigned int row, unsigned int column, ResponseLayer *src)
175  {
176  int scale = this->width / src->width;
177 
178 #ifdef RL_DEBUG
179  assert(src->getCoords(row, column) == this->getCoords(scale * row, scale * column));
180 #endif
181 
182  return responses[(scale * row) * width + (scale * column)];
183  }
184 
185 #ifdef RL_DEBUG
186  std::vector<std::pair<int, int>> coords;
187 
188  inline std::pair<int,int> getCoords(unsigned int row, unsigned int column)
189  {
190  return coords[row * width + column];
191  }
192 
193  inline std::pair<int,int> getCoords(unsigned int row, unsigned int column, ResponseLayer *src)
194  {
195  int scale = this->width / src->width;
196  return coords[(scale * row) * width + (scale * column)];
197  }
198 #endif
199  };
