76  inline T *createInstance(const std::string &name, const Data &data){
77  typename std::map<std::string,Plugin>::iterator it = plugins.find(name);
78  if(it == plugins.end()){
79  std::ostringstream all;
80  for(typename std::map<std::string,Plugin>::iterator jt = plugins.begin(); jt != plugins.end();){
81  all << jt->first;
82  if(++jt != plugins.end()) all << ",";
83  }
84  throw utils::ICLException("PluginRegister<T>: unknow type: '"+ name + "' (registered are: " +all.str() + ")");
85  return 0;
86  }
87  return it->second.create(data);
88  }
