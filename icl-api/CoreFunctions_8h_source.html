344  ICLCore_API inline void getMinAndMax(T a, T b, T c, T &minVal, T &maxVal){
345  if(a<b) {
346  minVal=a;
347  maxVal=b;
348  }
349  else {
350  minVal=b;
351  maxVal=a;
352  }
353  if(c<minVal)
354  minVal=c;
355  else {
356  maxVal = c>maxVal ? c : maxVal;
357  }
